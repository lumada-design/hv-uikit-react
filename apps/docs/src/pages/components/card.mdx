### Media

You can place an image on your card using the `HvCardMedia` component. It will adapt to the location of the card it is placed on.

```jsx live
<HvCard className="w-240px" bgcolor="atmo1">
  <HvCardMedia
    component="img"
    alt="Schedule New Job"
    height={120}
    image="https://s7d9.scene7.com/is/image/hitachivantara/take-charge-of-your-hybrid-cloud-estate-everflex-control-square:mediumvertical?fmt=webp"
  />
  <HvCardContent>
    <HvTypography>
      Use the live code editor to change the position of the `HvCardMedia`
      component in the card.
    </HvTypography>
  </HvCardContent>
</HvCard>
```

### Selectable

You can mark a card as selectable by adding the `selectable` prop. The `selected` prop indicates whether or not the card is selected.

```tsx live
import { useState } from "react";

export default function Demo() {
  const [checked, setChecked] = useState(null);

  return (
    <div className="flex flex-row gap-2 w-full">
      {storages.map((p, idx) => (
        <StorageCard
          selected={checked === idx}
          title={p.name}
          icon={p.icon}
          description={p.description}
          onSelect={() => setChecked(idx)}
        />
      ))}
    </div>
  );
}

function StorageCard({
  selected,
  title,
  icon,
  description,
  onSelect,
  ...others
}: StorageCardProps) {
  return (
    <HvCard
      bgcolor="atmo1"
      className="flex-1"
      selectable
      selected={selected}
      {...others}
    >
      <button
        type="button"
        onClick={onSelect}
        aria-label={`Press enter or space to select the ${title}.`}
        className="w-full text-left"
      >
        <HvCardHeader
          title={
            <div className="flex justify-between items-center">
              <div className="flex gap-1 items-center">
                {icon}
                <HvTypography variant="title4">{title}</HvTypography>
              </div>
              <HvRadio
                onChange={onSelect}
                checked={selected}
                value="value"
                aria-label={`Tick to select the ${title} payment.`}
              />
            </div>
          }
        />
        <HvCardContent>{description}</HvCardContent>
      </button>
    </HvCard>
  );
}

interface StorageCardProps extends HvCardProps {
  title: string;
  icon: React.ReactNode;
  description: string;
  onSelect?: () => void;
}

const storages = [

    name: "Local Storage",

  },
  {
    name: "Cloud Storage",
    icon: <Cloud size="S" />,
    description: "Use cloud storage for secure, scalable data access.",
  },
];
```
