### Controlled

Controlled radio button group. Choosing the first option will result in an invalid state of the Radio group.

```tsx live
import { useState } from "react";

export default function Demo() {
  const [value, setValue] = useState("2");
  const [status, setStatus] = useState<HvFormStatus>("standBy");

  return (
    <HvCheckBoxGroup
      label="Choose the best radio button"
      value={value}
      status={status}
      statusMessage="Don\'t select 'None'!"
      onChange={(evt, newValue) => {
        setValue(newValue);
        setStatus(newValue.includes("none") ? "invalid" : "valid");
      }}
    >
      <HvCheckBox label="None" value="none" />
      <HvCheckBox label="Radio 1" value="1" />
      <HvCheckBox label="Radio 2" value="2" />
    </HvCheckBoxGroup>
  );
}
```
