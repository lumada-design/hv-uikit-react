### Width

You can use the `width` prop to pass an array of custom breakpoints for the step navigation.

```tsx live
<HvStepNavigation
  steps={[
    {
      title: "Completed",
      state: "Completed",
      onClick: () => {
        alert("You clicked on 'Completed' step");
      },
    },
    {
      title: "Failed",
      state: "Failed",
      onClick: () => {
        alert("You clicked on 'Failed' step");
      },
    },
    {
      title: "Pending",
      state: "Pending",
      onClick: () => {
        alert("You clicked on 'Pending' step");
      },
    },
    {
      title: "Current",
      state: "Current",
      onClick: () => {
        alert("You clicked on 'Current' step");
      },
    },
    {
      title: "Enabled",
      state: "Enabled",
      onClick: () => {
        alert("You clicked on 'Enabled' step");
      },
    },
    {
      title: "Disabled",
      state: "Disabled",
      onClick: () => {
        alert("You clicked on 'Disabled' step");
      },
    },
  ]}
  width={{
    xs: 200,
    sm: 400,
    md: 600,
    lg: 800,
    xl: 1000,
  }}
/>
```

### Custom styles

```tsx live
<div className="w-full grid place-content-center">
  <HvStepNavigation
    type="Simple"
    steps={[
      { title: "Import data" },
      { title: "Transform data" },
      { title: "Validate data" },
      { title: "Export data" },
    ].map((s) => ({
      ...s,
      className: "bg-positive",
      separatorClassName: "bg-positive h-3px",
    }))}
  />
</div>
```
