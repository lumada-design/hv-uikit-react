### Root component

The `HvTypography` component is a polymorphic component. Its default component is `p`,
but it can be changed to any other element through the `component` prop.

```tsx live
<div className="flex flex-col gap-sm">
  <HvTypography component="span">Typography as a span</HvTypography>
  <HvTypography
    component="a"
    href="https://lumada-design.github.io/uikit/master"
    link
  >
    Typography as a link
  </HvTypography>
</div>
```

You can also change the root component of the `HvTypography` component to a
custom component.

```tsx live
interface CustomLinkProps extends HvTypographyProps<"a"> {
  to: string;
}

const CustomLink = ({ to, children, ...others }: CustomLinkProps) => (
  <a href={to} {...others}>
    {children}
  </a>
);

export default function Demo() {
  return (
    <HvTypography
      link
      component={CustomLink}
      to="https://lumada-design.github.io/uikit/master"
    >
      Custom link
    </HvTypography>
  );
}
```

### Color

You can use the [`color`](/docs/styling#color-prop) prop to style the color of the typography.

```tsx live
<div className="flex flex-col gap-sm">
  <HvTypography color="text">This is a typography</HvTypography>
  <HvTypography color="positive">This is a typography</HvTypography>
  <HvTypography color="coral">This is a typography</HvTypography>
  <HvTypography color="#c73aa8">This is a typography</HvTypography>
  <HvTypography link>This is a typography working as a link</HvTypography>
  <HvTypography link color="text">
    This is a typography working as a link
  </HvTypography>
</div>
```
