import {
  skeletonClasses as classes,
  HvSkeleton,
} from "@hitachivantara/uikit-react-core";

import { Header } from "../../components/Header";

<Header
  params={{ name: "Skeleton", classes }}
  Component={HvSkeleton}
  componentName="HvSkeleton"
  controls={{
    variant: {
      defaultValue: "text",
    },
    animation: {
      defaultValue: "wave",
    },
    hidden: {
      defaultValue: false,
    },
  }}
  decoratorClassName="w-190px"
  children={<div>My skeleton example content</div>}
/>

### Content size

If you pass no `width` or `height` props, the skeleton will adapt to the content size.

```tsx live
<div className="flex flex-col gap-xs">
  <HvSkeleton variant="circle">
    <HvAvatar alt="Beatrice" src="https://i.imgur.com/bE7vg3N.png" size="md" />
  </HvSkeleton>
  <HvSkeleton variant="text">
    <HvTypography>Some text being hidden by the skeleton</HvTypography>
  </HvSkeleton>
  <HvSkeleton variant="square">
    <HvTypography variant="title3">
      Some text being hidden by the skeleton
    </HvTypography>
  </HvSkeleton>
</div>
```

### Custom animation

You can apply custom CSS animations to `HvSkeleton` by defining them in your own styles.

```tsx live
<HvSkeleton
  width={300}
  height={100}
  variant="square"
  className="!animate-flash !animate-count-infinite"
/>
```
