import {
  stackClasses as classes,
  HvPanel,
  HvStack,
} from "@hitachivantara/uikit-react-core";

import { Header } from "../../components/Header";

<Header
  params={{ name: "Stack", classes }}
  Component={HvStack}
  componentName="HvStack"
  controls={{
    direction: { defaultValue: "row" },
    spacing: { defaultValue: "sm" },
    divider: { defaultValue: false },
  }}
>
  {[...Array(4).keys()].map((index) => (
    <HvPanel key={index}>{index + 1}</HvPanel>
  ))}
</Header>

### Custom direction

You can set the `direction` prop responsively based on breakpoints.  
For example, use a column layout on `xs` and `sm`, and switch to a row layout on `md` and above.

```tsx live
<HvStack direction={{ xs: "column", md: "row" }} spacing="xs" divider={false}>
  {[...Array(4).keys()].map((index) => (
    <HvPanel key={index}>{index + 1}</HvPanel>
  ))}
</HvStack>
```

### Custom divider

The `divider` prop accepts either a boolean—to render the default divider—or a `ReactNode` for a custom divider element.

```tsx live
<HvStack direction="row" className="flex items-center" divider={<DropRight />}>
  {[...Array(4).keys()].map((index) => (
    <HvPanel className="rounded-full size-48px grid place-content-center">
      {index + 1}
    </HvPanel>
  ))}
</HvStack>
```
