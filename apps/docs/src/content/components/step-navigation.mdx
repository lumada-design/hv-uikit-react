### Width

You can use the `width` prop to pass an array of custom breakpoints for the step navigation.

```tsx live
<HvStepNavigation
  steps={[
    {
      title: "Completed",
      state: "Completed",
    },
    {
      title: "Failed",
      state: "Failed",
    },
    {
      title: "Pending",
      state: "Pending",
    },
    {
      title: "Current",
      state: "Current",
    },
    {
      title: "Enabled",
      state: "Enabled",
    },
    {
      title: "Disabled",
      state: "Disabled",
    },
  ]}
  width={{ xs: 200, sm: 400, md: 600, lg: 800, xl: 1000 }}
/>
```

### Custom styles

```tsx live
<div className="w-full grid place-content-center">
  <HvStepNavigation
    type="Simple"
    steps={[
      { title: "Import data" },
      { title: "Transform data" },
      { title: "Validate data" },
      { title: "Export data" },
    ].map((s) => ({
      ...s,
      className: "bg-positive",
      separatorClassName: "bg-positive h-3px",
    }))}
  />
</div>
```
