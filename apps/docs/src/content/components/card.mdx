### Media

Use the `HvCardMedia` component to add an image to your card. It automatically adapts to its placement within the card.

```tsx live
<HvCard className="w-340px" bgcolor="bgContainer">
  <HvCardMedia
    component="img"
    alt="Schedule New Job"
    height={120}
    image="https://s7d9.scene7.com/is/image/hitachivantara/take-charge-of-your-hybrid-cloud-estate-everflex-control-square:mediumvertical?fmt=webp"
  />
  <HvCardContent>
    <HvTypography>
      Use the live code editor to change the position of the `HvCardMedia`
      component in the card.
    </HvTypography>
  </HvCardContent>
</HvCard>
```

### Status

The `HvCard` component can have a status color on the top. Use the `statusColor` prop to set it with any [color](/docs/styling#color-prop).

```tsx live
<HvCard className="w-440px" bgcolor="bgContainer" statusColor="negative">
  <HvCardHeader title={<HvTypography variant="title4">Error</HvTypography>} />
  <HvCardContent>
    <HvTypography>Something went wrong. Please try again later.</HvTypography>
  </HvCardContent>
  <HvActionBar>
    <HvButton variant="primarySubtle">Dismiss</HvButton>
  </HvActionBar>
</HvCard>
```

The status can also be represented with an icon by using the `icon` prop.

```tsx live
<HvCard
  className="w-440px"
  bgcolor="bgContainer"
  icon={<HvStatusIcon variant="error" weight="duotone" type="simple" />}
>
  <HvCardHeader title={<HvTypography variant="title4">Error</HvTypography>} />
  <HvCardContent>
    <HvTypography>
      Something went wrong. Please try again later.
    </HvTypography>
  </HvCardContent>
  <HvActionBar>
    <HvButton variant="primarySubtle">Dismiss</HvButton>
  </HvActionBar>
</HvCard>
```

### Selectable

Make a card selectable by using the `selectable` prop. Use the `selected` prop to control its selection state.

```tsx live
import { useState } from "react";
import { Cloud, Storage } from "@hitachivantara/uikit-react-icons";

export default function Demo() {
  const [checked, setChecked] = useState(null);

  return (
    <div className="flex flex-row gap-sm w-full">
      {storages.map((p, idx) => (
        <StorageCard
          key={idx}
          selected={checked === idx}
          title={p.name}
          icon={p.icon}
          description={p.description}
          onSelect={() => setChecked(idx)}
        />
      ))}
    </div>
  );
}

function StorageCard({
  selected,
  title,
  icon,
  description,
  onSelect,
  ...others
}: StorageCardProps) {
  return (
    <HvCard
      bgcolor="bgContainer"
      className="flex-1"
      selectable
      selected={selected}
      {...others}
    >
      <button
        type="button"
        onClick={onSelect}
        aria-label={`Press enter or space to select the ${title}.`}
        className="w-full text-left"
      >
        <HvCardHeader
          title={
            <div className="flex justify-between items-center">
              <div className="flex gap-xs items-center">
                {icon}
                <HvTypography variant="title4">{title}</HvTypography>
              </div>
              <HvRadio
                onChange={onSelect}
                checked={selected}
                value="value"
                aria-label={`Tick to select the ${title} payment.`}
              />
            </div>
          }
        />
        <HvCardContent>{description}</HvCardContent>
      </button>
    </HvCard>
  );
}

interface StorageCardProps extends HvCardProps {
  title: string;
  icon: React.ReactNode;
  description: string;
  onSelect?: () => void;
}

const storages = [
  {
    name: "Local Storage",
    icon: <HvStatusIcon customIcon={<Storage size="S" />} />,
    description: "Use local storage for quick access to user settings.",
  },
  {
    name: "Cloud Storage",
    icon: <HvStatusIcon customIcon={<Cloud size="S" />} />,
    description: "Use cloud storage for secure, scalable data access.",
  },
];
```
