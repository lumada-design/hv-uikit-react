import { getComponentData } from "../../utils/component";

import { Playground } from "../../components/code/Playground";
import { Header } from "../../components/Header";

export const getStaticProps = async ({ params }) => {
  const meta = await getComponentData("Footer", "core", footerClasses);
  return { props: { ssg: { meta } } };
};

### Custom labels

You can pass custom content to the `name`, `copyright`, and `links` props.

```tsx live
<HvFooter
  className="w-full"
  name={<HvTypography variant="title4">Pentaho+</HvTypography>}
  copyright={
    <HvTypography variant="caption1">Get a Grip On Your Data</HvTypography>
  }
  links={
    <HvTypography link component="a" href="https://pentaho.com/">
      Visit us
    </HvTypography>
  }
/>
```
